interaction {
    id: SERIAL PRIMARY KEY
    user_id: VARCHAR(255) NOT NULL,
    event_type: VARCHAR(100) NOT NULL,
    metadata: JSONB NULL
    created_at: TIMESTAMP DEFAULT NOW()
    INDEX idx_user_id (user_id),
    INDEX idx_event_type (event_type)


}
